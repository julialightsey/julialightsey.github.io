--
-- sys.sysprocesses
-- https://msdn.microsoft.com/en-us/library/ms179881.aspx
-- Contains information about processes that are running on an instance of SQL Server. 
--  These processes can be client processes or system processes. 
-------------------------------------------------     
select [sysprocesses].[hostname]                 as [hostname],
       [databases].[name]                        as [database],
       [dm_exec_sql_text].[text]                 as [sql_text],
       [dm_exec_sql_text].[objectid]             as [sql_object_id],
       suser_sname([databases].[owner_sid])      as [database_owner],
       [sysprocesses].[loginame]                 as [process_user_name],
       [dm_exec_requests].[blocking_session_id]  as [blocking_session_id],
       [dm_exec_sessions].[login_time]           as [login_time],
       [dm_exec_sessions].[cpu_time]             as [cpu_time],
       [dm_exec_sessions].[memory_usage]         as [memory_usage],
       [dm_exec_sessions].[total_scheduled_time] as [total_scheduled_time],
       [dm_exec_sessions].[total_elapsed_time]   as [total_elapsed_time],
       [dm_exec_sessions].[reads]                as [reads],
       [dm_exec_sessions].[writes]               as [writes],
       [sysprocesses].*
from   sys.sysprocesses as [sysprocesses]
       cross apply sys.dm_exec_sql_text([sysprocesses].[sql_handle]) as [dm_exec_sql_text]
       left join sys.dm_exec_requests as [dm_exec_requests]
              on [dm_exec_requests].[sql_handle] = [sysprocesses].[sql_handle]
       join sys.databases as [databases]
         on [databases].[database_id] = [sysprocesses].[dbid]
       left join sys.dm_exec_sessions as [dm_exec_sessions]
              on [dm_exec_sessions].[session_id] = [dm_exec_requests].[session_id]
order  by [sysprocesses].[hostname],
          [databases].[name];

--
-- sys.dm_os_tasks
-- https://msdn.microsoft.com/en-us/library/ms174963(v=sql.110).aspx
-- Returns one row for each task that is active in the instance of SQL Server.
-------------------------------------------------     
select *
from   sys.dm_os_tasks as [dm_os_tasks];

--
-- sys.dm_os_threads
-- https://msdn.microsoft.com/en-us/library/ms187818(v=sql.110).asp
-- Returns a list of all SQL Server Operating System threads that are running under the SQL Server process.
-------------------------------------------------    
select *
from   sys.dm_os_threads as [dm_os_threads];

/* 
--
-- sys.dm_exec_requests
-- https://msdn.microsoft.com/en-us/library/ms177648.aspx
-- Returns information about each request that is executing within SQL Server.
-- 
-- sys.dm_exec_sessions 
-- https://msdn.microsoft.com/en-us/library/ms176013.aspx
--
  Returns one row per authenticated session on SQL Server. sys.dm_exec_sessions is a 
  server-scope view that shows information about all active user connections and 
  internal tasks. This information includes client version, client program name, client 
  login time, login user, current session setting, and more. Use sys.dm_exec_sessions 
  to first view the current system load and to identify a session of interest, and then 
  learn more information about that session by using other dynamic management views or 
  dynamic management functions.
  The sys.dm_exec_connections, sys.dm_exec_sessions, and sys.dm_exec_requests dynamic management 
  views map to the sys.sysprocesses system table. 
*/
-------------------------------------------------    
select *
from   sys.dm_exec_requests as [dm_exec_requests]
       left join sys.dm_exec_sessions as [dm_exec_sessions]
              on [dm_exec_sessions].[session_id] = [dm_exec_requests].[session_id];
