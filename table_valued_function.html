use [chamomilesql];

go

--
-------------------------------------------------
if schema_id(N'kate_test') is null
  execute (N'create schema kate_test');

go

--
-------------------------------------------------
if object_id(N'[kate_test].[table_valued_function]', N'TF') is not null
  drop function [kate_test].[table_valued_function];

go

create function [kate_test].[table_valued_function](@object [sysname])
returns @objectlist table (
  [name]        [sysname],
  [type_desc]   [sysname],
  [object_id]   [int],
  [create_date] [datetime])
as
  begin
      insert into @objectlist
                  ([name],[type_desc],[object_id],[create_date])
      select [name],
             [type_desc],
             [object_id],
             [create_date]
      from   sys.objects
      where  [name] = @object;

      return;
  end;

go

if object_id(N'[kate_test].[sample_01]', N'U') is not null
  drop table [kate_test].[sample_01];

go

create table [kate_test].[sample_01]
  (
     [id]    [int] identity(1, 1) not null,
     [color] [sysname]
  );

go

--
-------------------------------------------------
select [name],
       [type_desc],
       [object_id],
       [create_date]
from   [kate_test].[table_valued_function] (N'sample_01');

go
